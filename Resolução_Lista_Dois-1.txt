    {
        delta = pow(variavelB, 2) - 4 * variavelA * variavelC;
        if (delta < 0){
                printf("Nao existe raiz, tente novamente");
        }else if (delta == 0){
            variavelx1 = -variavelB / (2 * variavelA);
            printf("Raiz: %.2f", variavelx1);

            variavelx1=(-variavelB+sqrt(delta))/(2*variavelA);
            variavelx2=(-variavelB-sqrt(delta))/(2*variavelA);
            printf("\nRaiz primeira Linha: %.2f e \nRaiz primeira Linha: %.2f", variavelx1, variavelx2);
